						exists = owner.py:military_party

				value = 1.5

			}
		}

			limit = {
				or = {
					law_stance = {
						law = law_type:law_professional_army
						value > approve
					}
					law_stance = {
						law = law_type:law_private_army
						value > approve
					}
				}
			}
			multiply = {
				value = 2
				desc = "from_loves_the_army"
			}
		}

			limit = {
				and = {

					}

				desc = "from_dislikes_the_army"

			limit = {
				or = {
					law_stance = {
						law = law_type:law_professional_army
						value < disapprove
