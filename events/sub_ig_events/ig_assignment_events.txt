############################# ARMED FORCES #############################
sub_ig_assignment.1 = { # Juntaists
	type = country_event
	hidden = yes
	trigger = {
		not = { has_variable = juntaists_leader }
	}
	immediate = {
		create_character = {
			interest_group = ig_armed_forces
			on_created = {
				set_variable = juntaists_trigger
				owner = {
					set_variable = {
						name = juntaists_leader
						value = PREV
					}
				}
			}		
		}
	}
}
