namespace = sub_ig_assignment
#	Assigns leader to SUB-IGs
############################# ARMED FORCES #############################
sub_ig_assignment.1 = { # Juntaists
	type = country_event
	hidden = yes
	trigger = {
		or = {
			not = { has_variable = juntaists_leader }
			and = {
				has_variable = juntaists_leader
				not = {
					any_scope_character = { has_variable = juntaists_leader_check }
				}
			}
		}

	}
	immediate = {
		create_character = {
			interest_group = ig_armed_forces
			save_scope_as = juntaists_leader_scope
		}
		
		scope:juntaists_leader_scope = {
			place_character_in_void = 60
			set_variable = juntaists_leader_check
		}
		
		set_variable = {
			name = juntaists_leader
			value = scope:juntaists_leader_scope
		}
	}
}
